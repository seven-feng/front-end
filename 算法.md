## 排序

### 二分排序

~~~ javascript
function binarySearch(array, key) {
    var low = 0, high = array.length, mid;
    while(low <= high) {
        mid = (low + high) / 2;
        if(key < array[mid]) {
            high = mid - 1;
        } else if(key > array[mid]) {
            low = mid + 1;
        } else {
            return mid;
        }
    }
    return -1;
}
~~~

### 快速排序

~~~ javascript
function quickSort(array, left, right) {
    if(left < right) {
        var index = patition(array,left,right);
        quickSort(array,left,index-1);
        quickSort(array,index+1,right);
    }
}

function patition(array,left,right) {
    var begin = left, end = right, key = array[right];
    while(begin < end) {
        while(begin < end && array[begin] <= key) {
            begin++;
        }
        while(begin < end && array[end] >= key) {
            end--;
        }
        var temp = array[end];
        array[end] = array[begin];
        array[begin] = temp;
    }
    var temp = array[right];
    array[right] = array[begin];
    array[begin] = temp;
    return begin;
}

var a = [1,2,3,5,4];
quickSort(a,0,4);
~~~

``` javascript
function bubbleSort(array) {
    var len = array.length;
    for(var i = 0; i < len-1; i++) {
        for(var j = 0; j < len-1-i; j++) {
            if(array[j] > array[j+1]) {
                swap(array[j],array[j+1]);
            }
        }
    }
}
```

### 选择排序

``` javascript
function selectionSort(array) {
    var len = array.length;
    for(var i = 0; i < len-1; i++) {
        var min = i;
        for(var j = i+1; j < len; j++) {
            if(array[j] < array[min]) {
                min = j;
            }
        }
        swap(array[i],array[min]);
    }
}
```

``` javascript
function insertionSort(array) {
    var len = array.length;
    for(var i = 1; i < len; i++) {
        for(var j=i; j > 0; j--) {
            if(array[j] < array[j-1]) {
                swap(array[j], array[j-1])
            } 
        }
    }
}
```

### 希尔排序

``` javascript
function shellSort(array) {
    var len = array.length;
    var gap = [5,3,1];
    for(g = 0; g < gap.length; g++) {
        for(var i = gap[g]; i < len; i++) {
            for(var j=i; j >= gap[g]; j=j-gap[g]) {
                if(array[j] < array[j-gap[g]]) {
                    swap(array[j], array[j-gap[g]])
                } 
            }
        }
    }
}
```

## 动态规划

``` javascript
function lcs(word1,word2) {
    var len1 = word1.length;
    var len2 = word2.length;
    var lcs = new Array(len1+1);
    for(var i = 0; i <= len1; i++) {
        lcs[i] = new Array(len2+1);
        for(var j = 0; j < len2; j++) {
            lcs[i][j] = 0;
        }
    }
    var max = 0, index = -1;
    for(i = 0; i <= len1; i++) {
        for(j = 0; j <= len2; j++) {
            if(i==0 || j==0) {
                lcs[i][j] = 0;
            }
            if(word1[i] == word2[j]) {
                lcs[i][j] = lcs[i-1][j-1] + 1;
            } else {
                lcs[i][j] = 0;
            }
            if(lcs[i][j] > max) {
                index = i;
                max = lcs[i][j];
            }
        }
    }
    var str = "";
    for(i = max-index; i < index; i++) {
        str += word1[i];
    }
}
```



### 背包

``` javascript
function dknapsack(capacity,size,value,n) {
    var k = [];
    for(var i = 0; i <= n; i++) {
        k[i] = [];
    }

    for(var i = 0; i <= n; i++) {
        for(var w = 0; w <= capacity; w++) {
            if(i == 0 || w == 0) {
                k[i][w] = 0;
            } else if(w >= size[i-1]) {
                k[i][w] = max(k[i-1][w-size[i-1]] + value[i-1],k[i-1][w]);
            } else {
                k[i][w] = k[i-1][w];
            }
        }
    }
}
```
